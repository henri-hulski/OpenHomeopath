210:     /**
211:      * get_result_data retrieves the repertorization data.
212:      *
213:      * get_result_data takes the selected symptoms and retrieves from the database the correspodending remedies and grades and count them.
214:      *
215:      * @return void
216:      * @access private
217:      */
218:     private function get_result_data() {
219:         global $db;
220:         $query = "DROP TEMPORARY TABLE IF EXISTS result";
221:         $db->send_query($query);
222:         $query = "CREATE TEMPORARY TABLE result (
223:             rem_id smallint(4) unsigned NOT NULL,
224:             points mediumint(8) unsigned NOT NULL,
225:             hits mediumint(8) unsigned NOT NULL,
226:             PRIMARY KEY(rem_id)
227:             ) ENGINE=MEMORY DEFAULT CHARSET=utf8";
228:         $db->send_query($query);
229:         foreach ($this->sym_select as $sym_id => $degree) {
230:             $sym_id_query_ar[] = $this->sym_rem_tbl . ".sym_id = $sym_id";
231:             $query = "SELECT rem_id, MAX(grade) FROM " . $this->sym_rem_tbl . " WHERE sym_id = $sym_id GROUP BY sym_id, rem_id ORDER BY NULL";
232:             $result = $db->send_query($query);
233:             while(list($rem_id, $grade) = $db->db_fetch_row($result)) {
234:                 $points = $grade * $degree;
235:                 $query = "INSERT INTO result SET rem_id = $rem_id, points = $points, hits = 1 ON DUPLICATE KEY UPDATE points = points + $points, hits = hits + 1";
236:                 $db->send_query($query);
237:             }
238:             $db->free_result($result);
239:         }
240:         $sym_id_query = implode(" OR ", $sym_id_query_ar);
241:         $query = "SELECT remedies.rem_short, " . $this->sym_rem_tbl . ".grade, " . $this->sym_rem_tbl . ".sym_id FROM " . $this->sym_rem_tbl . ", remedies WHERE " . $this->sym_rem_tbl . ".rem_id = remedies.rem_id AND ($sym_id_query) ORDER BY remedies.rem_short, " . $this->sym_rem_tbl . ".grade DESC, " . $this->sym_rem_tbl . ".sym_id ASC";
242:         $db->send_query($query);
243:         while($rel = $db->db_fetch_row()) {
244:             $this->rel_ar[] = $rel;
245:         }
246:         $db->free_result();
247:         $query = "SELECT result.points, result.hits, remedies.rem_short, remedies.rem_name, remedies.rem_id FROM result, remedies WHERE remedies.rem_id = result.rem_id ORDER BY result.points DESC, result.hits DESC, remedies.rem_short ASC";
248:         $db->send_query($query);
249:         $this->rel_count = 0;
250:         $this->remedies_ar = array();
251:         while($rem = $db->db_fetch_row()) {
252:             $this->remedies_ar[] = $rem;
253:             $this->rel_count += $rem[1];
254:         }
255:         $db->free_result();
256:         $query = "SELECT COUNT(*) FROM result";
257:         $db->send_query($query);
258:         list($this->rem_count) = $db->db_fetch_row();
259:         $db->free_result();
260:         $query = "DROP TEMPORARY TABLE result";
261:         $db->send_query($query);
262:         $this->sym_count = count($this->sym_select);
263:     }

266:      * get_symptoms_ar builds a sorted symptoms array.
267:      *
268:      * @return void
269:      * @access private

271:     private function get_symptoms_ar() {
272:         global $db, $session;
273:         $query = "DROP TEMPORARY TABLE IF EXISTS sym_sort";
274:         $db->send_query($query);
275:         $query = "CREATE TEMPORARY TABLE sym_sort (
276:             id mediumint(8) unsigned NOT NULL,
277:             name varchar(510) NOT NULL,
278:             degree tinyint(1) unsigned NOT NULL,
279:             kuenzli tinyint(1) unsigned NOT NULL,
280:             PRIMARY KEY(id)
281:             ) ENGINE=MEMORY DEFAULT CHARSET=utf8";
282:         $db->send_query($query);
283:         foreach ($this->sym_select as $sym_id => $degree) {
284:             $lang = $session->lang;
285:             $query = "SELECT " . $this->symptoms_tbl . ".symptom, main_rubrics.rubric_$lang, sym_src.kuenzli FROM (" . $this->symptoms_tbl . ", main_rubrics) LEFT JOIN sym_src ON sym_src.sym_id = " . $this->symptoms_tbl . ".sym_id WHERE main_rubrics.rubric_id = " . $this->symptoms_tbl . ".rubric_id AND " . $this->symptoms_tbl . ".sym_id = $sym_id";
286:             $db->send_query($query);
287:             list ($symptom, $main_rubric, $kuenzli) = $db->db_fetch_row();
288:             $db->free_result();
289:             $full_name = $db->escape_string($main_rubric . " >> " . $symptom);
290:             $kuenzli = (empty($kuenzli)) ? 0 : $kuenzli;
291:             $query = "INSERT INTO sym_sort VALUES ($sym_id, '$full_name', $degree, $kuenzli)";
292:             $db->send_query($query);
293:         }
294:         $query = "SELECT * FROM sym_sort ORDER BY degree DESC, name ASC";
295:         $db->send_query($query);
296:         for ($sym_ar = array(); $row = $db->db_fetch_assoc(); $sym_ar[array_shift($row)] = $row);
297:         $db->free_result();
298:         $query = "DROP TEMPORARY TABLE sym_sort";
299:         $db->send_query($query);
300:         $this->symptoms_ar = $sym_ar;
301:     }

383:     /**
384:      * get_sym_rem_src returns the sources that exists for a given symptom-remedy-relation.
385:      *
386:      * @param integer $sym_id         Symptom ID
387:      * @param integer $rem_id         Remedy ID
388:      * @param integer &$max_grade     The max. grade of the symptom-remedy-relation
389:      * @param integer &$kuenzli_dot   0|1 if the symptom-remedy-relation has a Künzli-dot 1 otherwise 0.
390:      * @return string
391:      * @access private
392:      */
393:     private function get_sym_rem_src($sym_id, $rem_id, &$max_grade, &$kuenzli_dot) {
394:         global $db;
395:         $sources = "";
396:         $kuenzli_dot = 0;
397:         $max_grade = $this->get_max_grade($sym_id, $rem_id);
398:         $query = "SELECT sr.src_id, sr.grade, sr.rel_id, sr.kuenzli, ss.status_symbol FROM " . $this->sym_rem_tbl . " sr, sym_status ss WHERE sr.sym_id = $sym_id AND sr.rem_id = $rem_id AND ss.status_id = sr.status_id ORDER BY ss.status_grade DESC, sr.grade DESC, sr.src_id ASC";
399:         $result = $db->send_query($query);
400:         while (list($src_id, $grade, $rel_id, $kuenzli, $status_symbol) = $db->db_fetch_row($result)) {
401:             $source = "$src_id";
402:             if ($grade != $max_grade) {
403:                 $source .= "($grade" . _("-gr.") . ")";
404:             }
405:             $query = "SELECT src_id, nonclassic FROM sym_rem_refs WHERE rel_id = $rel_id ORDER BY nonclassic, src_id";
406:             $db->send_query($query);
407:             unset($ref_array);
408:             while (list ($ref_id, $nonclassic) = $db->db_fetch_row()) {
409:                 $ref = "$ref_id";
410:                 if ($nonclassic == 1) {
411:                     $ref .= "_(nk)";
412:                 }
413:                 $ref_array[] = $ref;
414:             }
415:             $db->free_result();
416:             if (!empty($ref_array)) {
417:                 $refs = implode(",", $ref_array);
418:                 $source .= ":$refs";
419:             }
420:             if (!empty($status_symbol)) {
421:                 $source .= $status_symbol;
422:             }
423:             if ($kuenzli == 1) {
424:                 $kuenzli_dot = 1;
425:             }
426:             $sources_ar[] = $source;
427:         }
428:         $db->free_result($result);
429:         if (!empty($sources_ar)) {
430:             $sources =  implode(" / ", $sources_ar);
431:         }
432:         return $sources;
433:     }
