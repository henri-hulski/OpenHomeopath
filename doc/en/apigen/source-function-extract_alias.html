 30: function parse_express_script ($sym_rem, $src_id, $lang_id, $rubric_id) {
 31: 
 32:     global $text;
 33:     $prev_symptom = "";
 34:     create_temporary_express_tables();
 35:     $sym_rem = str_replace("\r", "\n", $sym_rem);
 36:     $sym_rem = preg_replace("/\n[\n\s]*\n/u", "\n", $sym_rem);
 37:     $sym_rem = str_replace("%", "&percent;", $sym_rem);
 38:     $sym_rem = str_replace("\n", "%", $sym_rem);
 39:     if ($sym_rem{strlen($sym_rem)-1} == "%") {
 40:         $sym_rem = substr_replace($sym_rem, "", -1, 1);
 41:     }    // % at the end will be deleted
 42:     $sym_rem = preg_replace("/\s+/u", ' ', $sym_rem);  // delete whitespace
 43:     $sym_rem = explode("%", $sym_rem);  // put the script lines in an array
 44:     foreach ($sym_rem as $key => $value) {
 45:         $value = str_replace("&percent;", "%", $value);
 46:         if (strpos($value, ":")) {  // check for a colon
 47:             list($symptom, $remedy) = explode(":", $value, 2);
 48:             $symptom = trim ($symptom);
 49:             if ($symptom != "alias" && $symptom != "source" && $symptom != "ref" && !empty($src_id)) {  // check if we don't have an alias or source/reference record
 50:                 $sympt_id = extract_symptom($symptom, $rubric_id, $lang_id);
 51:                 if ($sympt_id != 0) {
 52:                     if (preg_match("/\{[\s,]*(.+)\}[\s,]*/u", $remedy, $matches)) {
 53:                         extract_remedies($matches[1], $sympt_id, 1);
 54:                     }
 55:                     $remedy = preg_replace("/\s*\{.*\}\s*/u", "", $remedy);
 56:                     if ($remedy != "") {
 57:                         extract_remedies($remedy, $sympt_id, 0);
 58:                     }
 59:                 } else {
 60:                     $text .= $symptom . ": " . $remedy . "\n";
 61:                 }
 62:             } elseif ($symptom == "alias") {  // if alias record
 63:                 extract_alias($remedy);
 64:             } elseif ($symptom == "source" || $symptom == "ref") {  // if source/reference record
 65:                 if ($symptom == "source") {
 66:                     $primary_src = 1;
 67:                 } else {
 68:                     $primary_src = 0;
 69:                 }
 70:                 $extracted_source = extract_source($remedy, $primary_src);
 71:                 if ($extracted_source["error"] == 0) {
 72:                     unset($extracted_source["error"]);
 73:                     $source_ar[] = $extracted_source;
 74:                 }
 75:             } else {
 76:                 $text .= $symptom . ": " . $remedy . "\n";
 77:                 $nq++;
 78:             }
 79:         } else {
 80:             $text .= $value . "\n";
 81:             $prev_symptom = "";
 82:             $nn++;
 83:         }
 84:     }
 85: }  // end function parse_express_script
